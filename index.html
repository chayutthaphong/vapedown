    <script>

        // ==========================================

        // CONFIGURATION: PASTE YOUR WEB APP URL HERE

        // ==========================================

        const GAS_URL = "https://script.google.com/macros/s/AKfycbzlhn9cGOixGFZip6BGwXeYpL9M1DU5I920DxELbgp3h7S3myPQVJvsx10xcQhpe5P-RQ/exec"; 


        // Local State Management

        let currentTarget = 10;

        let chartInstance = null;


        // Initialize elements

        const datePicker = document.getElementById('datePicker');

        const targetSelect = document.getElementById('targetSelect');

        const currentPuffsText = document.getElementById('currentPuffs');

        const dailyTargetDisplay = document.getElementById('dailyTargetDisplay');

        const motivationMsg = document.getElementById('motivationMsg');

        const logBtn = document.getElementById('logBtn');

        const syncStatus = document.getElementById('syncStatus');

        const syncDot = document.getElementById('syncDot');


        // Set default date to today (Local Time)

        const todayStr = new Date().toISOString().split('T')[0];

        datePicker.value = todayStr;


        // Load data from LocalStorage

        function getLogHistory() {

            const history = localStorage.getItem('vape_log_history_v2');

            return history ? JSON.parse(history) : {};

        }


        function saveLogHistory(history) {

            localStorage.setItem('vape_log_history_v2', JSON.stringify(history));

        }


        // UI Refresh Logic

        function refreshUI() {

            const selectedDate = datePicker.value;

            const history = getLogHistory();

            const todayCount = history[selectedDate] || 0;

            

            currentTarget = parseInt(targetSelect.value);

            

            currentPuffsText.innerText = todayCount;

            dailyTargetDisplay.innerText = currentTarget;


            // Motivation & Feedback Logic

            if (todayCount === 0) {

                motivationMsg.innerText = "A fresh start. Keep your focus sharp today.";

                motivationMsg.className = "text-xs text-gray-500 mt-3 px-4 font-light h-4";

            } else if (todayCount < currentTarget / 2) {

                motivationMsg.innerText = "Excellent control. Pacing is highly disciplined.";

                motivationMsg.className = "text-xs text-gray-600 mt-3 px-4 font-medium h-4";

            } else if (todayCount < currentTarget) {

                motivationMsg.innerText = `${currentTarget - todayCount} puffs remaining. Practice delay tactics.`;

                motivationMsg.className = "text-xs text-amber-600 mt-3 px-4 font-medium h-4";

            } else if (todayCount === currentTarget) {

                motivationMsg.innerText = "Benchmark reached. Stop here to secure today's win.";

                motivationMsg.className = "text-xs text-green-600 mt-3 px-4 font-semibold h-4";

            } else {

                motivationMsg.innerText = `Over limit by ${todayCount - currentTarget}. Reset your rhythm tomorrow.`;

                motivationMsg.className = "text-xs text-red-500 mt-3 px-4 font-medium h-4";

            }


            updateChart();

        }


        // Get past 7 days dates array for chart

        function getLast7Days() {

            const dates = [];

            for (let i = 6; i >= 0; i--) {

                const d = new Date();

                d.setDate(d.getDate() - i);

                dates.push(d.toISOString().split('T')[0]);

            }

            return dates;

        }


        // Format date string

        function formatMinDate(dateStr) {

            const options = { month: 'short', day: 'numeric' };

            return new Date(dateStr).toLocaleDateString('en-US', options);

        }


        // Chart.js Rendering Logic

        function updateChart() {

            const ctx = document.getElementById('analyticsChart').getContext('2d');

            const last7Days = getLast7Days();

            const history = getLogHistory();


            const actualData = last7Days.map(date => history[date] || 0);

            const benchmarkData = last7Days.map(() => parseInt(targetSelect.value));

            const labels = last7Days.map(date => formatMinDate(date));


            if (chartInstance) {

                chartInstance.destroy();

            }


            chartInstance = new Chart(ctx, {

                type: 'bar',

                data: {

                    labels: labels,

                    datasets: [

                        {

                            label: 'Actual Puffs',

                            data: actualData,

                            backgroundColor: '#111827',

                            borderRadius: 4,

                            barPercentage: 0.4,

                            order: 2

                        },

                        {

                            label: 'Benchmark Line',

                            data: benchmarkData,

                            type: 'line',

                            borderColor: '#9ca3af',

                            borderWidth: 1.5,

                            borderDash: [3, 3],

                            pointRadius: 0,

                            fill: false,

                            order: 1

                        }

                    ]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    plugins: {

                        legend: { display: false }

                    },

                    scales: {

                        x: {

                            grid: { display: false },

                            ticks: { font: { size: 9 }, color: '#9ca3af' }

                        },

                        y: {

                            grid: { color: '#f9fafb' },

                            ticks: { beginAtZero: true, font: { size: 9 }, color: '#9ca3af', stepSize: 5 }

                        }

                    }

                }

            });

        }


        // Event Listeners

        datePicker.addEventListener('change', refreshUI);

        targetSelect.addEventListener('change', refreshUI);


        logBtn.addEventListener('click', () => {

            const selectedDate = datePicker.value;

            const history = getLogHistory();

            

            // Increment

            history[selectedDate] = (history[selectedDate] || 0) + 1;

            saveLogHistory(history);

            refreshUI();


            // Real-time timestamp

            const now = new Date();

            const timeStr = now.toTimeString().split(' ')[0];

            const currentStatus = (history[selectedDate] <= currentTarget) ? "On Track" : "Over Limit";


            // Sync UI

            syncStatus.innerText = "Syncing to Google Sheets...";

            syncDot.className = "w-2 h-2 rounded-full bg-amber-500 animate-pulse";


            if (GAS_URL === "YOUR_WEB_APP_URL_HERE") {

                console.warn("GAS URL not configured.");

                syncStatus.innerText = "Saved locally. (GAS URL missing)";

                syncDot.className = "w-2 h-2 rounded-full bg-gray-400";

                return;

            }


            const formData = new URLSearchParams();

            formData.append('date', selectedDate);

            formData.append('time', timeStr);

            formData.append('count', 1);

            formData.append('target', currentTarget);

            formData.append('status', currentStatus);


            fetch(GAS_URL, {

                method: 'POST',

                body: formData,

                mode: 'no-cors'

            })

            .then(() => {

                syncStatus.innerText = "Cloud backup complete";

                syncDot.className = "w-2 h-2 rounded-full bg-green-500";

            })

            .catch(err => {

                console.error("Sync failed:", err);

                syncStatus.innerText = "Sync failed. Will retry next puff.";

                syncDot.className = "w-2 h-2 rounded-full bg-red-500";

            });

        });


        // Initialize

        refreshUI();
